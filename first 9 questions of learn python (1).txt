//Assignment 5.3: WAP to add all prime numbers from 1 to n and n is given by the user.

def is_prime(num):
    if num <= 1:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True
n = int(input("Enter the value of n: "))
sum_of_primes = 0
for i in range(1, n + 1):
    if is_prime(i):
        sum_of_primes += i
print("The sum of all prime numbers from 1 to", n, "is:", sum_of_primes)
